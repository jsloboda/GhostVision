1function [ data_reconstd ] = gtdata( data )
  2%GTDATA Perform Gabor Transform on data to extract hearbeat frequency range
  3
  4%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  5%
  6%  Algorithm
  7%  1. get recorded data
  8%  2. pass filtered data into Gabor Transform function
  9%  3. plot:
 10%     (1) dowsampled data
 11%     (2) magnitude of Gabor coefficients
 12%     (3) magnitude of SELECTED gabor coefficients
 13%     (4) reconstructed signal, using selected coefficients
 14%     (5) reconstructed signal, using all coefficients
 15%
 16%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 17
 18%% initialize simulation parameters
 19Fs = 100; % Hz
 20
 21
 22
 23%% get data
 24
 25%load('sampledata.mat')
 26% down-sample: keep every 80th sample to go from 8000->100 Hz
 27%data = downsample(data3, 80);
 28% eliminate first 100 samples (motion artifacts in certain data files)
 29%data = data(100:end);
 30
 31%figure('Name','signal');
 32%plot(data)
 33
 34
 35%% Gabor Transform Data
 36
 37a = 1;    %length of time shift for dgt/idgt, must be positive integer
 38M = 1000; %number of channels for dgt/idgt. 
 39          % frequency resolution = Fs/M = 100/1000 = 0.1   
 40         
 41[data_gt_coeffs, Ls] = dgt(data,'gauss',a,M);
 42% Gabor Transform (from ltfat toolbox)
 43%    DGT(f,g,a,M) computes the Gabor coefficients of the input
 44%    signal f with respect to the window g and parameters a and M. The
 45%    output is a vector/matrix in a rectangular layout.
 46%    Input parameters:
 47%          f     : Input data
 48%          g     : Window function.
 49%          a     : Length of time shift.
 50%          M     : Number of channels.
 51%          L     : Length of transform to do.
 52%   Output parameters:
 53%          c     : M*N array of coefficients.
 54%          Ls    : Length of input signal.
 55%
 56%    References:
 57%    K. Gröchenig. Foundations of Time-Frequency Analysis. Birkhäuser, 
 58%    2001.
 59%      
 60%    H. G. Feichtinger and T. Strohmer, editors. Gabor Analysis and
 61%    Algorithms. Birkhäuser, Boston, 1998.
 62%     
 63%      Demos:  demo_dgt
 64
 65%% plot coefficients
 66
 67%figure('Name', 'complete transform coefficients')
 68%plotdgt(data_gt_coeffs, a, M);
 69% data_gt_coeffs [1000x1000] complex, 
 70% 0 to 500 "hz" (each row is .1 hz), 1000 time bins (10 sec, Fs = 100)
 71% use row 8 for 0.65 to 0.75
 72% rows 1-500 are positive freq, rows 501-1000 are negative freq
 73% use row 994 also for -0.65 to -0.75 Hz 
 74
 75% plotdgt(coef,a) plots the Gabor coefficients coef. The coefficients
 76% must have been produced with a time shift of a.
 77% 
 78% plotdgt(coef,a,fs) does the same assuming a sampling rate of fs Hz 
 79% of the original signal.
 80
 81
 82%% choose coefficients for between 8/12 and 5/6 Hz  (.6666 - .714...)
 83c = zeros(size(data_gt_coeffs));
 84c(8,:) = data_gt_coeffs(8,:);
 85c(994,:) = data_gt_coeffs(994,:);
 86%m = freq, n = time for c(m,n)
 87
 88%figure('Name', 'selected transform coefficients')
 89%plotdgt(c, 1, Fs);
 90%ylim([-2 2])
 91
 92%% reconstruct a signal using just those coefficients
 93
 94%figure('Name','partial reconstructed signal');
 95data_reconstd=idgt(c,'gauss',a,Ls);
 96%plot(data_reconstd/100);
 97
 98%% reconstruct a signal using all coefficients
 99%figure('Name','complete reconstructed signal');
100data_reconstd_full=idgt(data_gt_coeffs,'gauss',a,Ls);
101%plot(data_reconstd_full/100);
102
103% idgt(c,g,a) computes the Gabor expansion of the input coefficients c 
104% with respect to the window g and time shift a. The number of channels 
105% is deduced from the size of the coefficients c.
106
107%% unused: spectrogram of signal
108%figure('Name','spectrogram');
109%data_sgram=sgram(data,Fs,'lin');
110%title('Spectrogram of the data signal');
111
112%    SGRAM(f,sr) plots a spectrogram of f using a DGT for a signal with 
113%    sampling rate sr (sampled with sr samples per second)
114
115
116
117end
